# Warcraft
# Sacrifice Occupied Province Character

character_event = {
	id = SSI.61
	hide_window = yes
	
	ai = yes

	# Warcraft
	# min_age = 16

	prisoner = no
	
	is_triggered_only = yes
	
	trigger = {

		# Warcraft
		can_be_sacrificed_trigger = yes

		in_command = no

		# Warcraft
		NOT = { same_realm = new_character }
		new_character = { religion_have_sacrifices_trigger = yes }

	}
	
	immediate = {

		# Warcraft
		liege = {
			if = {
				limit = { ai = no }

				character_event = { id = SSI.62 }
			}
		}
		new_character = {
			set_character_flag = wc_do_not_show_post_sacrifice_ping_event_flag
			ROOT = {
				caltulate_PREV_offmap_old_gods_currency_insanity_effect = yes
				add_offmap_old_gods_currency_insanity_after_calculation_effect = yes
				PREV_public_sacrifice_THIS_effect = yes
			}
			clr_character_flag = wc_do_not_show_post_sacrifice_ping_event_flag
		}

	}
	
	option = {
		name = OK
	}
}
